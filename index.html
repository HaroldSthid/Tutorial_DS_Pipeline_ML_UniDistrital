def exploracion_rapida(df, nombre="Dataset"):
    """Función todo-en-uno para explorar datasets"""
    print(f"\n{'='*50}")
    print(f"📊 ANÁLISIS DE: {nombre}")
    print(f"{'='*50}")
    
    # 1. Dimensiones
    print(f"\n📏 Dimensiones: {df.shape[0]} filas × {df.shape[1]} columnas")
    
    # 2. Memoria utilizada
    memoria = df.memory_usage(deep=True).sum() / 1024**2
    print(f"💾 Memoria: {memoria:.2f} MB")
    
    # 3. Calidad de datos
    nulos = df.isnull().sum()
    print(f"\n🔍 Calidad de datos:")
    print(f"   - Valores nulos totales: {nulos.sum():,}")
    print(f"   - Columnas con nulos: {(nulos > 0).sum()}")
    
    # 4. Duplicados
    duplicados = df.duplicated().sum()
    print(f"   - Filas duplicadas: {duplicados:,}")
    
    # 5. Estadísticas rápidas
    numeric_cols = df.select_dtypes(include=[np.number]).columns
    if len(numeric_cols) > 0:
        print(f"\n📈 Top 5 columnas numéricas:")
        print(df[numeric_cols[:5]].describe().round(2))
    
    # 6. Detectar outliers (método IQR)
    for col in numeric_cols[:3]:
        Q1 = df[col].quantile(0.25)
        Q3 = df[col].quantile(0.75)
        IQR = Q3 - Q1
        outliers = ((df[col] < (Q1 - 1.5 * IQR)) | (df[col] > (Q3 + 1.5 * IQR))).sum()
        if outliers > 0:
            print(f"⚠️ {col}: {outliers} outliers detectados")
    
    return df

def limpiar_moneda_avanzado(valor):
    """
    Convierte cualquier formato monetario a número float
    Maneja: $1,234,567 | 1.234.567 | COP 1,234,567 | etc.
    """
    if pd.isna(valor):
        return 0.0
    if isinstance(valor, (int, float)):
        return float(valor)
    
    # Convertir a string y limpiar
    valor_str = str(valor).strip().upper()
    
    # Eliminar texto no numérico
    for simbolo in ['$', 'COP', 'USD', 'EUR', ',', '.', ' ', '(', ')']:
        valor_str = valor_str.replace(simbolo, '')
    
    # Extraer solo dígitos y punto decimal
    valor_limpio = ''.join(c for c in valor_str if c.isdigit() or c == '.')
    
    try:
        return float(valor_limpio) if valor_limpio else 0.0
    except:
        return 0.0

🔬 Snippet 4 - K-Means con selección automática de k
def kmeans_optimizado(df, feature_cols, max_k=10):
    """
    K-Means con selección automática de k usando método del codo
    y visualización de clusters
    """
    # Preparar datos
    X = df[feature_cols].fillna(df[feature_cols].mean())
    
    # Normalización (crítica para K-Means)
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)
    
    # Método del codo con mejora visual
    inertias = []
    k_range = range(1, max_k + 1)
    
    for k in k_range:
        kmeans = KMeans(n_clusters=k, random_state=42, n_init=10)
        kmeans.fit(X_scaled)
        inertias.append(kmeans.inertia_)
    
    # Encontrar el "codo" automáticamente
    diffs = np.diff(inertias)
    diffs_2 = np.diff(diffs)
    k_optimo = np.argmin(diffs_2) + 2
    
    print(f"🎯 k óptimo sugerido: {k_optimo}")
    
    # Aplicar k óptimo
    kmeans_final = KMeans(n_clusters=k_optimo, random_state=42, n_init=10)
    clusters = kmeans_final.fit_predict(X_scaled)
    
    # Análisis de clusters
    df_temp = df[feature_cols].copy()
    df_temp['Cluster'] = clusters
    
    print(f"\n📊 Tamaño de clusters:")
    print(df_temp['Cluster'].value_counts().sort_index())
    
    print(f"\n📈 Centros de clusters (normalizados):")
    print(pd.DataFrame(kmeans_final.cluster_centers_, 
                       columns=feature_cols).round(3))
    
    return clusters, kmeans_final, scaler

def comparar_clasificadores(X_train, X_test, y_train, y_test):
    """
    Compara múltiples clasificadores y muestra el mejor
    """
    from sklearn.linear_model import LogisticRegression
    from sklearn.svm import SVC
    from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
    
    clasificadores = {
        'Árbol de Decisión': DecisionTreeClassifier(max_depth=4, random_state=42),
        'Random Forest': RandomForestClassifier(n_estimators=100, random_state=42),
        'Gradient Boosting': GradientBoostingClassifier(n_estimators=100, random_state=42),
        'SVM': SVC(kernel='rbf', random_state=42)
    }
    
    resultados = []
    
    for nombre, clf in clasificadores.items():
        clf.fit(X_train, y_train)
        y_pred = clf.predict(X_test)
        acc = accuracy_score(y_test, y_pred)
        resultados.append({'Clasificador': nombre, 'Precisión': acc})
        
        print(f"{nombre}: {acc:.2%}")
    
    # Mostrar mejor modelo
    mejor = max(resultados, key=lambda x: x['Precisión'])
    print(f"\n🏆 MEJOR MODELO: {mejor['Clasificador']} con {mejor['Precisión']:.2%}")
    
    return pd.DataFrame(resultados).sort_values('Precisión', ascending=False)

from sklearn.model_selection import GridSearchCV

def optimizar_hyperparametros(X_train, y_train):
    """
    Búsqueda automática de mejores hiperparámetros
    """
    param_grid = {
        'max_depth': [3, 4, 5, 6, 8],
        'min_samples_split': [5, 10, 20, 50],
        'min_samples_leaf': [2, 4, 8, 16],
        'criterion': ['gini', 'entropy']
    }
    
    grid_search = GridSearchCV(
        DecisionTreeClassifier(random_state=42),
        param_grid,
        cv=5,  # Validación cruzada 5-fold
        scoring='accuracy',
        n_jobs=-1,
        verbose=1
    )
    
    grid_search.fit(X_train, y_train)
    
    print(f"🎯 Mejores parámetros: {grid_search.best_params_}")
    print(f"📊 Mejor score CV: {grid_search.best_score_:.2%}")
    
    return grid_search.best_estimator_

def extraer_reglas_arbol(tree, feature_names, class_names):
    """
    Extrae reglas legibles de un árbol de decisión
    """
    tree_ = tree.tree_
    feature_name = [
        feature_names[i] if i != -2 else "undefined!"
        for i in tree_.feature
    ]
    
    reglas = []
    
    def recurse(node, depth, condition):
        if tree_.feature[node] != -2:
            name = feature_name[node]
            threshold = tree_.threshold[node]
            
            # Rama izquierda
            left_condition = condition + [f"{name} <= {threshold:.2f}"]
            recurse(tree_.children_left[node], depth + 1, left_condition)
            
            # Rama derecha
            right_condition = condition + [f"{name} > {threshold:.2f}"]
            recurse(tree_.children_right[node], depth + 1, right_condition)
        else:
            # Nodo hoja
            class_id = np.argmax(tree_.value[node])
            clase = class_names[class_id]
            condicion_str = " Y ".join(condition)
            reglas.append(f"SI {condicion_str} ENTONCES → {clase}")
    
    recurse(0, 0, [])
    return reglas

def metricas_avanzadas(y_true, y_pred, model=None, X_test=None):
    """
    Calcula métricas avanzadas de clasificación
    """
    from sklearn.metrics import roc_auc_score, log_loss, f1_score, precision_recall_curve
    
    print("📊 MÉTRICAS DE EVALUACIÓN")
    print("="*50)
    
    # Métricas básicas
    print(f"🎯 Accuracy: {accuracy_score(y_true, y_pred):.2%}")
    print(f"🎯 F1 Score (macro): {f1_score(y_true, y_pred, average='macro'):.2%}")
    
    # Si es clasificación binaria y tenemos probabilidades
    if model and hasattr(model, 'predict_proba') and len(np.unique(y_true)) == 2:
        y_proba = model.predict_proba(X_test)[:, 1]
        print(f"🎯 AUC-ROC: {roc_auc_score(y_true, y_proba):.2%}")
        print(f"🎯 Log Loss: {log_loss(y_true, y_proba):.4f}")
    
    # Matriz de confusión detallada
    cm = confusion_matrix(y_true, y_pred)
    print("\n📋 Matriz de Confusión:")
    print(cm)
    
    # Métricas por clase
    print("\n📊 Métricas por Clase:")
    print(classification_report(y_true, y_pred))
    
    # Valor económico del modelo (ejemplo)
    tn, fp, fn, tp = cm.ravel()
    costo_fp = 1000  # Costo de un falso positivo
    costo_fn = 5000  # Costo de un falso negativo
    beneficio = tp * 10000 - fp * costo_fp - fn * costo_fn
    print(f"\n💰 Beneficio estimado del modelo: ${beneficio:,.0f}")
    
    return {
        'accuracy': accuracy_score(y_true, y_pred),
        'f1_score': f1_score(y_true, y_pred, average='macro')
    }

🎯 Snippet 10 - Ejecución completa del pipeline
def pipeline_completo(df):
    """
    Ejecuta todo el flujo de análisis en una función
    """
    print("🚀 INICIANDO PIPELINE COMPLETO")
    print("="*60)
    
    # Paso 1: Crear variable objetivo
    print("\n1. Creando variable objetivo...")
    df['VALOR_TOTAL'] = np.random.uniform(50_000_000, 500_000_000, len(df))
    
    # Paso 2: Segmentación
    percentiles = {
        'p40': df['VALOR_TOTAL'].quantile(0.40),
        'p80': df['VALOR_TOTAL'].quantile(0.80)
    }
    
    def clasificar(v):
        if v <= percentiles['p40']: return 'Bajo'
        elif v <= percentiles['p80']: return 'Medio'
        else: return 'Alto'
    
    df['SEGMENTO'] = df['VALOR_TOTAL'].apply(clasificar)
    
    # Paso 3: Features para modelo
    np.random.seed(42)
    df['ANTIGUEDAD'] = np.random.randint(1, 365, len(df))
    df['FRECUENCIA'] = np.random.choice([1, 2, 3, 4, 5], len(df))
    
    features = ['VALOR_TOTAL', 'ANTIGUEDAD', 'FRECUENCIA']
    
    # Paso 4: Entrenamiento
    X = df[features].fillna(df[features].mean())
    y = df['SEGMENTO']
    
    le = LabelEncoder()
    y_encoded = le.fit_transform(y)
    
    X_train, X_test, y_train, y_test = train_test_split(
        X, y_encoded, test_size=0.2, random_state=42, stratify=y_encoded
    )
    
    model = DecisionTreeClassifier(max_depth=4, random_state=42)
    model.fit(X_train, y_train)
    
    # Paso 5: Evaluación
    y_pred = model.predict(X_test)
    accuracy = accuracy_score(y_test, y_pred)
    
    print(f"\n✅ Precisión del modelo: {accuracy:.2%}")
    
    # Paso 6: Exportar resultados
    df.to_csv('resultado_final.csv', index=False)
    from google.colab import files
    files.download('resultado_final.csv')
    
    print("\n🎉 PIPELINE COMPLETADO!")
    return model, df
